output_continuous <- function(x) {

  # Calculate test statistic
  test <- t_test(
    x = df,
    formula = as.formula(x %+% " ~ self_id"),
    order = c("TRUE", "FALSE") # Doesn't do anything important, but suppresses a warning
  )

  # Format output
  out <- tibble(
    statistic = "t(" %+% round(test$t_df) %+% ") = " %+% round(test$statistic, 2),
    p = test$p_value
  )

  return(out)
  
}

# Function to analyze ordinal variables
output_ordinal <- function(x) {

  # Calculate test statistic
  test <- wilcox.test(
    as.numeric(df[[x]]) ~ df$self_id, 
    na.rm = T, 
    paired = F, 
    exact = F
  )
  
  # Format output
  out <- tibble(
    statistic = "U = " %+% round(test$statistic, 2),
    p = test$p.value
  )

  return(out)
  
}

Equivalence Testing

eq_test <- function(x) {
  
  id_mean = mean(df[[x]][df$self_id], na.rm = T)
  dx_mean = mean(df[[x]][!df$self_id], na.rm = T)
  
  id_sd = sd(df[[x]][df$self_id], na.rm = T)
  dx_sd = sd(df[[x]][!df$self_id], na.rm = T)
  
  id_n = sum(!is.na(df[[x]][df$self_id]))
  dx_n = sum(!is.na(df[[x]][!df$self_id]))
  
  # This is the same as looking at the 90% CI and seeing if it overlaps a given bound...
  TOSTER::tsum_TOST(
    m1 = id_mean, m2 = dx_mean, sd1 = id_sd, sd2 = dx_sd, n1 = id_n, n2 = dx_n,
    hypothesis = "EQU",
    paired = F,
    var.equal = T,
    eqb = .3,
    alpha = .05, # This doesn't affect the test, only the provided CI
    eqbound_type = "SMD"
  )
  
}

q5_cols <- df %>%
  select(starts_with("q5")) %>%
  names()

q5_comparison <- tibble()

for(x in q5_cols) {
  
  table <- table(df$self_id, df[[x]])
  
  sum_self_id <- sum(df[[x]][df$self_id], na.rm = T)
  mean_self_id <- percent(mean(df[[x]][df$self_id], na.rm = T))
  
  sum_dx <- sum(df[[x]][!df$self_id], na.rm = T)
  mean_dx <- percent(mean(df[[x]][!df$self_id], na.rm = T))

  chisq <- chisq.test(table)
  
  expected <- chisq$expected
  
  if(min(expected) >= 5) {
    
    statistic <- chisq$statistic
    p_value <- chisq$p.value
    
  } else {
    
    statistic <- NA
    p_value <- NA
    
  }
  
  new_row <- tibble(
    var = x,
    self_id = sum_self_id %+% " (" %+% mean_self_id %+% ")",
    dx = sum_dx %+% " (" %+% mean_dx %+% ")",
    statistic = statistic,
    p_value = p_value
  )
  
  q5_comparison <- bind_rows(q5_comparison, new_row)
  
}
